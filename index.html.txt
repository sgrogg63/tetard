    // Horloge en direct
    function updateTime() {
      const now = new Date();
      const h = String(now.getHours()).padStart(2, '0');
      const m = String(now.getMinutes()).padStart(2, '0');
      document.getElementById('time').textContent = h + ":" + m;
    }
    setInterval(updateTime, 1000);
    updateTime();
